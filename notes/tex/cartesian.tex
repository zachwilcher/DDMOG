\section{Graph Products}

\begin{definition}
The Cartesian (box) product of two directed graphs \(\Gamma_1\) and \(\Gamma_2\) is a directed graph \(\Gamma_1 \square \Gamma_2\)
with vertex set \(V(\Gamma_1) \times V(\Gamma_2)\) where
\((\alpha, \beta)\) is directed at \((\alpha',\beta')\) if and only if exactly one of the following holds:
\begin{enumerate}
    \item \(\alpha = \alpha'\) and \(\beta\) is directed at \(\beta'\).
    \item \(\beta = \beta'\) and \(\alpha\) is directed at \(\alpha'\).
\end{enumerate}
\end{definition}

\begin{proposition}
Suppose \(\Gamma_1\) and \(\Gamma_2\) are \(G_1\)-DDMOG and \(G_2\)-DDMOG respectively.
If \(\Gamma_1\) and \(\Gamma_2\) are \textit{balanced},
then \(\Gamma_1 \square \Gamma_2\) is an \((G_1 \oplus G_2)\)-DDMOG.
\end{proposition}

\begin{proof}
Define a labeling \(\ell: V(\Gamma_1) \times V(\Gamma_2) \rightarrow G_1 \oplus G_2\)
by \(\ell(\alpha, \beta) = (\ell_1(\alpha), \ell_2(\beta))\)
where \(\ell_1: V(\Gamma_1) \rightarrow G_1\) and \(\ell_2: V(\Gamma_2) \rightarrow G_2\) are the labelings of \(\Gamma_1\) and \(\Gamma_2\) respectively.

Let \((\alpha, \beta) \in V(\Gamma_1 \square \Gamma_2)\).
Observe that the in-neighborhood and out-neighborhood of \((\alpha, \beta)\) are exactly the following.
\[
N^+(\Gamma_1 \square \Gamma_2, (\alpha, \beta)) =
\{(\alpha', \beta) : \alpha' \in N^+(\Gamma_1, \alpha)\} \cup \{(\alpha, \beta') : \beta' \in N^+(\Gamma_2, \beta)\}
\]
\[
N^-(\Gamma_1 \square \Gamma_2, (\alpha, \beta)) =
\{(\alpha', \beta) : \alpha' \in N^-(\Gamma_1, \alpha)\} \cup \{(\alpha, \beta') : \beta' \in N^-(\Gamma_2, \beta)\}
\]
We can then write the weight of \((\alpha, \beta)\) as follows.
\begin{align*}
\hspace{.66cm}
&
 \sum_{\mathclap{\alpha '\in N^+(\Gamma_1, \alpha)}} \ell(\alpha', \beta)
+ \sum_{\mathclap{\beta' \in N^+(\Gamma_2, \beta)}} \ell(\alpha, \beta')
- \sum_{\mathclap{\alpha' \in N^-(\Gamma_1, \alpha)}} \ell(\alpha', \beta)
- \sum_{\mathclap{\beta' \in N^-(\Gamma_2, \beta)}} \ell(\alpha, \beta') \\
=&
\sum_{\mathclap{\alpha '\in N^+(\Gamma_1, \alpha)}}   (\ell_1(\alpha'), \ell_2(\beta))
- \sum_{\mathclap{\alpha' \in N^-(\Gamma_1, \alpha)}} (\ell_1(\alpha'), \ell_2(\beta))
+ \sum_{\mathclap{\beta' \in N^+(\Gamma_2, \beta)}}   (\ell_1(\alpha),  \ell_2(\beta'))
- \sum_{\mathclap{\beta' \in N^-(\Gamma_2, \beta)}}   (\ell_1(\alpha),  \ell_2(\beta')) \\
=&
\sum_{\mathclap{\alpha '\in N^+(\Gamma_1, \alpha)}}   (\ell_1(\alpha'), 0)
- \sum_{\mathclap{\alpha' \in N^-(\Gamma_1, \alpha)}} (\ell_1(\alpha'), 0)
+ \sum_{\mathclap{\beta' \in N^+(\Gamma_2, \beta)}}   (0,  \ell_2(\beta'))
- \sum_{\mathclap{\beta' \in N^-(\Gamma_2, \beta)}}   (0,  \ell_2(\beta')) \\
=&
(k_1, 0) + (0, k_2) = (k_1, k_2)
\end{align*}

The second to last line follows from the fact that \(\Gamma_1\) and \(\Gamma_2\) are balanced.
Also, \(k_1\) and \(k_2\) are the magic constants of \(\Gamma_1\) and \(\Gamma_2\) respectively.



\end{proof}
